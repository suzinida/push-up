function calcAngle(a, b, c) {
  const ab = { x: a.x - b.x, y: a.y - b.y };
  const cb = { x: c.x - b.x, y: c.y - b.y };
  const dot = ab.x * cb.x + ab.y * cb.y;
  const mag = Math.sqrt(ab.x**2 + ab.y**2) * Math.sqrt(cb.x**2 + cb.y**2);
  if (mag === 0) return 180;
  return Math.acos(Math.max(-1, Math.min(1, dot / mag))) * (180 / Math.PI);
}

// 푸시업 자세 판별: 손이 바닥에 있고, 몸이 가로로 뻗어있음
function isPushupPose(lm) {
  const wristY = Math.max(lm[15].y, lm[16].y);
  if (wristY < 0.45) return false; // 손이 화면 아래쪽에 있어야 함
  const avgShY = (lm[11].y + lm[12].y) / 2;
  if (avgShY >= wristY - 0.03) return false; // 어깨는 손목보다 위에
  // 가로로 뻗은 몸통
  const avgShX = (lm[11].x + lm[12].x) / 2;
  const avgAnkleX = (lm[27].x + lm[28].x) / 2;
  if (Math.abs(avgShX - avgAnkleX) < 0.18) return false;
  return true;
}

// 보이는 쪽 선택 (visibility 평균이 높은 쪽)
function pickSide(lm) {
  const L = (lm[11].visibility + lm[13].visibility + lm[15].visibility + lm[23].visibility + lm[25].visibility) / 5;
  const R = (lm[12].visibility + lm[14].visibility + lm[16].visibility + lm[24].visibility + lm[26].visibility) / 5;
  return L >= R ? 'L' : 'R';
}

const pose = new Pose({ locateFile: (f) => `https://cdn.jsdelivr.net/npm/@mediapipe/pose/${f}` });
pose.setOptions({
  modelComplexity: 1, smoothLandmarks: true, enableSegmentation: false,
  minDetectionConfidence: 0.6, minTrackingConfidence: 0.6
});

pose.onResults((results) => {
  canvasEl.width  = videoEl.videoWidth  || 640;
  canvasEl.height = videoEl.videoHeight || 480;
  ctx.clearRect(0, 0, canvasEl.width, canvasEl.height);
  ctx.drawImage(results.image, 0, 0, canvasEl.width, canvasEl.height);

  if (!results.poseLandmarks) {
    statusEl.textContent = '사람을 찾는 중...';
    statusEl.className = 'overlay-status';
    feedbackEl.textContent = '준비';
    feedbackEl.style.color = 'var(--muted)';
    depthBarEl.style.height = '0%';
    return;
  }

  const lm = results.poseLandmarks;
  drawConnectors(ctx, lm, POSE_CONNECTIONS, { color: '#1f3a66', lineWidth: 1.5 });
  drawLandmarks(ctx, lm, { color: '#00d4ff', lineWidth: 1, radius: 3 });

  if (!isPushupPose(lm)) {
    statusEl.textContent = '팔굽혀펴기 자세를 취해주세요';
    statusEl.className = 'overlay-status';
    feedbackEl.textContent = '준비';
    feedbackEl.style.color = 'var(--muted)';
    sideEl.textContent = '자세 감지 중';
    depthBarEl.style.height = '0%';
    elbowAngleEl.textContent = '—';
    elbowStatEl.textContent = '—';
    bellyStateEl.textContent = '—';
    kneeStateEl.textContent = '—';
    setPhase('ready');
    return;
  }

  // 우세한 쪽(측면) 선택
  const side = pickSide(lm);
  const sh = side === 'L' ? lm[11] : lm[12];
  const el = side === 'L' ? lm[13] : lm[14];
  const wr = side === 'L' ? lm[15] : lm[16];
  const hp = side === 'L' ? lm[23] : lm[24];
  const kn = side === 'L' ? lm[25] : lm[26];
  sideEl.textContent = (side === 'L' ? '왼쪽' : '오른쪽') + ' 측면 인식';

  // 바닥 기준 = 손목 y (좀 더 안정적으로 양쪽 손목 최댓값)
  const floorY = Math.max(lm[15].y, lm[16].y);

  // 팔꿈치 각도 (어깨-팔꿈치-손목)
  const elbowAng = Math.round(calcAngle(sh, el, wr));
  elbowAngleEl.textContent = elbowAng;
  elbowStatEl.textContent = elbowAng;

  // 배(엉덩이) & 무릎 바닥과의 간격
  const bellyGap = floorY - hp.y;  // 양수 = 배가 바닥보다 위 (뜬 상태)
  const kneeGap  = floorY - kn.y;  // 양수 = 무릎이 바닥보다 위 (뜬 상태)

  const bellyOnFloor = Math.abs(bellyGap) < BELLY_ON_FLOOR;
  const kneeOnFloor  = Math.abs(kneeGap)  < KNEE_ON_FLOOR;
  const kneeOffFloor = kneeGap > KNEE_OFF_FLOOR;
  const elbowExt     = elbowAng >= ELBOW_EXTENDED;

  bellyStateEl.textContent = bellyOnFloor ? '닿음' : '뜸';
  kneeStateEl.textContent  = kneeOnFloor  ? '닿음' : (kneeOffFloor ? '뜸' : '중간');

  // DEPTH 표시 — DOWN에 가까울수록(배가 바닥에 가까울수록) 가득 참
  const depthPct = Math.round(Math.max(0, Math.min(100, (1 - Math.min(0.15, Math.abs(bellyGap)) / 0.15) * 100)));
  depthBarEl.style.height = depthPct + '%';

  // 단계 판정
  const isDown = bellyOnFloor && kneeOnFloor;
  const isUp   = elbowExt && kneeOffFloor;

  // 색상 / 피드백
  if (isDown) {
    depthBarEl.style.background = 'var(--accent2)';
    feedbackEl.textContent = 'DOWN!';
    feedbackEl.style.color = 'var(--accent2)';
    statusEl.textContent = '배와 무릎이 바닥에 닿았어요!';
    statusEl.className = 'overlay-status down';
  } else if (isUp) {
    depthBarEl.style.background = 'var(--accent)';
    feedbackEl.textContent = 'UP!';
    feedbackEl.style.color = 'var(--accent)';
    statusEl.textContent = '팔을 쭉 펴고 무릎이 떴어요!';
    statusEl.className = 'overlay-status up';
  } else if (depthPct > 60) {
    depthBarEl.style.background = 'var(--warn)';
    feedbackEl.textContent = 'GO DOWN';
    feedbackEl.style.color = 'var(--warn)';
    statusEl.textContent = '조금만 더 내려가세요';
    statusEl.className = 'overlay-status warn';
  } else {
    depthBarEl.style.background = 'var(--border)';
    feedbackEl.textContent = 'PUSH-UP';
    feedbackEl.style.color = 'var(--text)';
    statusEl.textContent = '팔굽혀펴기 자세 인식됨';
    statusEl.className = 'overlay-status';
  }

  // ═════ State machine ═════
  // 시퀀스: UP → DOWN → UP = 1 REP
  // ready 상태일 땐 어느 쪽이든 먼저 감지되면 그쪽으로 진입
  if (phase === 'ready') {
    if (isUp) setPhase('up');
    else if (isDown) setPhase('down');
  } else if (phase === 'up') {
    if (isDown) setPhase('down');
  } else if (phase === 'down') {
    if (isUp) {
      setPhase('up');
      totalCount++;
      totalCountEl.textContent = totalCount;
      flashEl(totalCountEl);
      showFlash('rep');
    }
  }

  // 관절 그리기 (어깨·팔꿈치·손·배(엉덩이)·무릎)
  const drawPoint = (p, color, label) => {
    const x = p.x * canvasEl.width;
    const y = p.y * canvasEl.height;
    ctx.beginPath(); ctx.arc(x, y, 9, 0, Math.PI * 2);
    ctx.strokeStyle = color; ctx.lineWidth = 2.5; ctx.stroke();
    ctx.fillStyle = color;
    ctx.font = '13px Do Hyeon, sans-serif';
    if (label) ctx.fillText(label, x + 12, y - 6);
  };

  drawPoint(sh, '#00d4ff', '어깨');
  drawPoint(el, elbowExt ? '#00e676' : '#ff9d00', '팔꿈치 ' + elbowAng + '°');
  drawPoint(wr, '#eaf3ff', '손');
  drawPoint(hp, bellyOnFloor ? '#00e676' : '#ff4d8d', bellyOnFloor ? '배 TOUCH' : '배');
  drawPoint(kn, kneeOnFloor ? '#00e676' : (kneeOffFloor ? '#00d4ff' : '#ff9d00'),
            kneeOnFloor ? '무릎 TOUCH' : (kneeOffFloor ? '무릎 OFF' : '무릎'));

  // 바닥 라인 (손목 y 기준) 시각화
  const fy = floorY * canvasEl.height;
  ctx.strokeStyle = 'rgba(0,212,255,0.35)';
  ctx.setLineDash([6, 6]); ctx.lineWidth = 1.2;
  ctx.beginPath(); ctx.moveTo(0, fy); ctx.lineTo(canvasEl.width, fy); ctx.stroke();
  ctx.setLineDash([]);
});

async function startCamera() {
  cameraStarted = true;
  try {
    const stream = await navigator.mediaDevices.getUserMedia({
      video: { facingMode: { exact: 'environment' } }
    }).catch(() => navigator.mediaDevices.getUserMedia({ video: true }));
    videoEl.srcObject = stream; videoEl.play();
    const camera = new Camera(videoEl, {
      onFrame: async () => { await pose.send({ image: videoEl }); },
      width: 640, height: 480
    });
    camera.start();
    loadingEl.style.display = 'none';
  } catch (e) {
    loadingEl.innerHTML = '<div style="color:#ff1744;font-size:15px;text-align:center;padding:20px;font-family:Do Hyeon,sans-serif">카메라 권한이 필요해요<br><small style="color:#6b84a8;font-size:13px">브라우저 설정에서 카메라를 허용해주세요</small></div>';
    cameraStarted = false;
  }
}
